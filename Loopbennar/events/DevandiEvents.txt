#Unifying Soul Of Nahrei - Shared Religious Ability
country_event = {
	id = devandi.1
	title = devandi.1.t
	desc = devandi.1.d
	picture = PRAYING_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = devandi.1.a
		if = {
			limit = { religion = askaeorg }
			custom_tooltip = askaeorg_steppe_dev_tt
			tooltip = {
				add_province_modifier = {
					name = unifying_soul_of_nahrei_mod
					duration = 3650 #10 years
				}
			}
			hidden_effect = {
				every_owned_province = {
					limit = { has_terrain = steppe }
					add_province_modifier = {
						name = unifying_soul_of_nahrei_mod
						duration = 3650 #10 years
					}
				}
			}
		}
		else = {
			capital_scope = { 
				add_base_tax = 1 
				area = {
					limit = { owned_by = ROOT }
					add_province_modifier = {
						name = unifying_soul_of_nahrei_mod
						duration = 3650 #10 years
					}
				}
			}
		}
	}
}

#Seismic Sense - Mazhthramazh Religious Ability
country_event = {
	id = devandi.4
	title = devandi.4.t
	desc = devandi.4.d
	picture = MIRACLE_MONK_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = devandi.4.a
		if = {
			limit = {
				any_owned_province = { has_empty_adjacent_province = yes }
				num_of_colonists = 1
			}
			custom_tooltip = seismic_sense_uncolonized_tt
			hidden_effect = { set_country_flag = seismic_sense_flag }
		}
		else = {
			random_owned_province = {
				limit = {
					OR = {
						trade_goods = iron
						trade_goods = copper
						trade_goods = gold
					}
					NOT = { has_province_modifier = seismic_sense_mod }
				}
				add_province_modifier = {
					name = seismic_sense_mod
					duration = 5475 #15 years
				}
			}
		}
	}
}

#Raid! - Askaeorg Religious Ability
country_event = {
	id = devandi.6
	title = devandi.6.t
	desc = devandi.6.d
	picture = HORDE_ON_HORSEBACK_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = devandi.6.a
		custom_tooltip = devandi_raid_tt
		hidden_effect = {
			set_variable = {
				which = devandi_raid_var
				value = 0
			}
			every_owned_province = {
				every_neighbor_province = {
					limit = { 
						NOT = { owned_by = ROOT }
						owner = { NOT = { is_subject_of = ROOT } } 
					}
					remove_loot = {
						who = ROOT
						amount = 100
					}
					ROOT = {
						change_variable = {
							which = devandi_raid_var
							value = 1
						}
					}
					owner = {
						if = {
							limit = {
								NOT = { 
									religion = askaeorg
									has_opinion_modifier = {
										who = ROOT
										modifier = devandi_raided_op
									}
								}
							}
							add_opinion = {
								who = ROOT
								modifier = devandi_raided_op
							}
						}
					}
				}
			}
			add_manpower = -1
			while = {
				limit = {
					check_variable = {
						which = devandi_raid_var
						value = 1
					}
				}
				add_manpower = -0.2
				subtract_variable = {
					which = devandi_raid_var
					value = 1
				}
			}
		}
	}
}
