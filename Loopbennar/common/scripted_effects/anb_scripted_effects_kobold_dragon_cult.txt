	tooltip = { 
		add_country_modifier = {
			name = $COLOUR$_dragon_$SLOT$_$UPGRADE$

			duration = -1
		}
	}
	add_church_aspect = $COLOUR$_dragon_$SLOT$_$UPGRADE$
}

kobold_hoard_increase_at_target = {
	if = {
		limit = { has_country_flag = new_hoard }
		187 = {
			change_variable = { dragon_cult_hoard = $AMOUNT$ } 
			upgrade_kobold_hoard = yes
		} 
	}
	else = {
		189 = {
			change_variable = { dragon_cult_hoard = $AMOUNT$ } 
			upgrade_kobold_hoard = yes
		}
	}
}
